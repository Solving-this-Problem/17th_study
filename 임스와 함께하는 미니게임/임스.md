##민웅
```python
# 25757_임스와 함께하는 미니게임_ImsMinigame
import sys
input = sys.stdin.readline

games = {
    'Y': 2,
    'F': 3,
    'O': 4
}

N, G = input().split()
N = int(N)

max_user = games[G]
user_list = {}
count = 1
ans = 0

for _ in range(N):
    user = input()
    if user in user_list.keys():
        continue
    else:
        user_list[user] = 0
        count += 1
        if count == max_user:
            ans += 1
            count = 1

print(ans)
```

```
import sys
input = sys.stdin.readline

N, G = input().split()
player = [input() for _ in range(int(N))]
player = list(set(player))	# set 중복 제거

if G == 'Y' :	# 만약 게임 종류가 'Y'(윷놀이)라면, print(len(player))를 실행하여 참가 가능한 게임 횟수(참가자 수)를 출력
    print(len(player))
elif G == 'F' :	# 같은 그림 찾기
    print(len(player) // 2)
elif G == 'O' :	# 원카드
    print(len(player) // 3)
```
